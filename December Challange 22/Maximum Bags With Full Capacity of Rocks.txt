class Solution {
public:
    int maximumBags(vector<int>& capacity, vector<int>& rocks, int additionalRocks) {
        vector<int>v;
        int n=capacity.size();
        for(int i=0;i<n;i++){
            capacity[i]-=rocks[i];
        }
        sort(capacity.begin(),capacity.end());
        int cnt=0;
        for(int i=0;i<n;i++){
            if(capacity[i]==0){
                cnt++;
            }else if(capacity[i]<=additionalRocks){
                cnt++;
                additionalRocks-=capacity[i];
            }else{
                break;
            }
        }
        return cnt;
    }
};
